# PostgreSQL Schema Design Guide

## Naming Conventions

```
Tables:       snake_case, plural        (users, order_items, blog_posts)
Columns:      snake_case, singular      (first_name, created_at, is_active)
Primary keys: id                        (not user_id — the table provides context)
Foreign keys: referenced_table_id       (user_id, order_id, parent_category_id)
Indexes:      idx_{table}_{columns}     (idx_users_email, idx_orders_status_date)
Unique:       uniq_{table}_{columns}    (uniq_users_email)
Check:        chk_{table}_{column}      (chk_orders_amount_positive)
Foreign key:  fk_{table}_{ref_table}    (fk_orders_users)
```

## Table Templates

### Standard Table

```sql
CREATE TABLE users (
  id         BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  email      TEXT NOT NULL UNIQUE,
  name       TEXT NOT NULL,
  status     TEXT NOT NULL DEFAULT 'active'
             CHECK (status IN ('active', 'inactive', 'suspended')),
  metadata   JSONB NOT NULL DEFAULT '{}',
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Auto-update updated_at
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER users_updated_at
  BEFORE UPDATE ON users
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

### Multi-Tenant Table

```sql
CREATE TABLE documents (
  id         BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  tenant_id  UUID NOT NULL REFERENCES tenants(id),
  title      TEXT NOT NULL,
  content    TEXT,
  created_by UUID NOT NULL REFERENCES users(id),
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Tenant-first index for all queries
CREATE INDEX idx_documents_tenant ON documents (tenant_id);
CREATE INDEX idx_documents_tenant_created
  ON documents (tenant_id, created_at DESC);

-- RLS for tenant isolation
ALTER TABLE documents ENABLE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation ON documents
  FOR ALL USING (tenant_id = current_setting('app.current_tenant_id')::uuid);
```

### Junction Table (Many-to-Many)

```sql
CREATE TABLE post_tags (
  post_id BIGINT NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
  tag_id  BIGINT NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
  PRIMARY KEY (post_id, tag_id)
);

-- Index for reverse lookups (find posts by tag)
CREATE INDEX idx_post_tags_tag ON post_tags (tag_id);
```

### Soft Delete Pattern

```sql
CREATE TABLE records (
  id         BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  data       TEXT NOT NULL,
  deleted_at TIMESTAMPTZ,   -- NULL = active, timestamp = soft-deleted
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Partial index: only active records (most queries)
CREATE INDEX idx_records_active ON records (created_at DESC)
  WHERE deleted_at IS NULL;

-- View for convenience
CREATE VIEW active_records AS
  SELECT * FROM records WHERE deleted_at IS NULL;
```

### Enum vs Check Constraint vs Lookup Table

```sql
-- Option 1: CHECK constraint (simple, no joins, easy to change)
-- Best for: small fixed sets unlikely to change
CREATE TABLE orders (
  status TEXT NOT NULL CHECK (status IN ('pending', 'paid', 'shipped', 'delivered'))
);

-- Option 2: PostgreSQL ENUM (type-safe, but hard to modify)
-- Best for: truly fixed sets (avoid — hard to add values)
CREATE TYPE order_status AS ENUM ('pending', 'paid', 'shipped', 'delivered');
CREATE TABLE orders (
  status order_status NOT NULL DEFAULT 'pending'
);

-- Option 3: Lookup table (most flexible, requires join)
-- Best for: values that change, need metadata, or are user-defined
CREATE TABLE order_statuses (
  code        TEXT PRIMARY KEY,
  label       TEXT NOT NULL,
  sort_order  INT NOT NULL DEFAULT 0,
  is_terminal BOOLEAN NOT NULL DEFAULT false
);
INSERT INTO order_statuses VALUES
  ('pending', 'Pending', 1, false),
  ('paid', 'Paid', 2, false),
  ('shipped', 'Shipped', 3, false),
  ('delivered', 'Delivered', 4, true);

CREATE TABLE orders (
  status TEXT NOT NULL REFERENCES order_statuses(code) DEFAULT 'pending'
);
```

**Recommendation:** Use CHECK constraints for most cases. They're simple, fast (no join), and adding a new value is a single ALTER TABLE.

## Primary Key Strategies

### IDENTITY (Recommended for Internal IDs)

```sql
-- GENERATED ALWAYS is safer than GENERATED BY DEFAULT
-- (prevents accidental manual ID insertion)
CREATE TABLE users (
  id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY
);

-- If you need to manually set IDs (migrations, imports):
INSERT INTO users (id, name) OVERRIDING SYSTEM VALUE VALUES (999, 'Imported');
```

### UUID (Recommended for Public-Facing IDs)

```sql
CREATE TABLE api_keys (
  id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    BIGINT NOT NULL REFERENCES users(id),
  key_hash   TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
```

### Composite Key

```sql
-- For junction tables and natural keys
CREATE TABLE enrollments (
  student_id BIGINT REFERENCES students(id),
  course_id  BIGINT REFERENCES courses(id),
  enrolled_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  PRIMARY KEY (student_id, course_id)
);
```

## Foreign Key Patterns

### ON DELETE Actions

| Action | Behavior | Use When |
|--------|----------|----------|
| `RESTRICT` (default) | Prevent deletion of parent | Protecting important references |
| `CASCADE` | Delete child rows too | Dependent data (comments on a post) |
| `SET NULL` | Set FK to NULL | Optional references (assigned_to) |
| `SET DEFAULT` | Set FK to default value | Reassign to default category |
| `NO ACTION` | Like RESTRICT but deferred | Circular references |

```sql
-- Cascade: deleting a user deletes their posts
ALTER TABLE posts ADD CONSTRAINT fk_posts_users
  FOREIGN KEY (author_id) REFERENCES users(id)
  ON DELETE CASCADE;

-- Set null: deleting an assignee clears the assignment
ALTER TABLE tasks ADD CONSTRAINT fk_tasks_assignee
  FOREIGN KEY (assigned_to) REFERENCES users(id)
  ON DELETE SET NULL;

-- Restrict: can't delete a category that has products
ALTER TABLE products ADD CONSTRAINT fk_products_categories
  FOREIGN KEY (category_id) REFERENCES categories(id)
  ON DELETE RESTRICT;
```

## Partitioning

### Range Partitioning (Time-Series Data)

```sql
-- Parent table
CREATE TABLE events (
  id         BIGINT GENERATED ALWAYS AS IDENTITY,
  event_type TEXT NOT NULL,
  payload    JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
) PARTITION BY RANGE (created_at);

-- Monthly partitions
CREATE TABLE events_2025_01 PARTITION OF events
  FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');
CREATE TABLE events_2025_02 PARTITION OF events
  FOR VALUES FROM ('2025-02-01') TO ('2025-03-01');
CREATE TABLE events_2025_03 PARTITION OF events
  FOR VALUES FROM ('2025-03-01') TO ('2025-04-01');

-- Indexes on partitions (each partition gets its own)
CREATE INDEX idx_events_created ON events (created_at DESC);
CREATE INDEX idx_events_type ON events (event_type);

-- Default partition (catches rows that don't match any partition)
CREATE TABLE events_default PARTITION OF events DEFAULT;
```

### Auto-Create Monthly Partitions

```sql
-- Create next month's partition automatically
CREATE OR REPLACE FUNCTION create_monthly_partition()
RETURNS void AS $$
DECLARE
  next_month DATE := date_trunc('month', now() + INTERVAL '1 month');
  partition_name TEXT := 'events_' || to_char(next_month, 'YYYY_MM');
  start_date TEXT := to_char(next_month, 'YYYY-MM-DD');
  end_date TEXT := to_char(next_month + INTERVAL '1 month', 'YYYY-MM-DD');
BEGIN
  EXECUTE format(
    'CREATE TABLE IF NOT EXISTS %I PARTITION OF events FOR VALUES FROM (%L) TO (%L)',
    partition_name, start_date, end_date
  );
END;
$$ LANGUAGE plpgsql;
```

### List Partitioning (By Category)

```sql
CREATE TABLE logs (
  id       BIGINT GENERATED ALWAYS AS IDENTITY,
  level    TEXT NOT NULL,
  message  TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
) PARTITION BY LIST (level);

CREATE TABLE logs_error PARTITION OF logs FOR VALUES IN ('error', 'fatal');
CREATE TABLE logs_warn PARTITION OF logs FOR VALUES IN ('warn');
CREATE TABLE logs_info PARTITION OF logs FOR VALUES IN ('info', 'debug');
```

## Common Anti-Patterns

### 1. Entity-Attribute-Value (EAV)

```sql
-- ANTI-PATTERN: EAV table
CREATE TABLE attributes (
  entity_id INT,
  key       TEXT,
  value     TEXT   -- Everything is text!
);

-- BETTER: Use JSONB for flexible attributes
CREATE TABLE entities (
  id         BIGINT PRIMARY KEY,
  type       TEXT NOT NULL,
  attributes JSONB NOT NULL DEFAULT '{}'
);

CREATE INDEX idx_entities_attrs ON entities USING gin (attributes);
```

### 2. Polymorphic Foreign Keys

```sql
-- ANTI-PATTERN: commentable_type + commentable_id
CREATE TABLE comments (
  commentable_type TEXT,  -- 'post' or 'photo'
  commentable_id   INT    -- No FK constraint possible!
);

-- BETTER: Separate nullable FKs
CREATE TABLE comments (
  post_id  BIGINT REFERENCES posts(id),
  photo_id BIGINT REFERENCES photos(id),
  CHECK (
    (post_id IS NOT NULL AND photo_id IS NULL) OR
    (post_id IS NULL AND photo_id IS NOT NULL)
  )
);

-- OR: Separate junction tables
CREATE TABLE post_comments (
  comment_id BIGINT REFERENCES comments(id),
  post_id    BIGINT REFERENCES posts(id)
);
```

### 3. Storing Computed Values Without Triggers

```sql
-- ANTI-PATTERN: manual denormalization that drifts
ALTER TABLE orders ADD COLUMN item_count INT;

-- BETTER: Generated column (always accurate)
ALTER TABLE orders ADD COLUMN total_amount NUMERIC
  GENERATED ALWAYS AS (quantity * unit_price) STORED;

-- OR: Materialized view for complex aggregations
CREATE MATERIALIZED VIEW order_stats AS
  SELECT user_id, COUNT(*) AS order_count, SUM(total) AS total_spent
  FROM orders GROUP BY user_id;

CREATE UNIQUE INDEX idx_order_stats_user ON order_stats (user_id);
REFRESH MATERIALIZED VIEW CONCURRENTLY order_stats;
```

### 4. Over-Normalization

```sql
-- ANTI-PATTERN: separate table for 1:1 relationship with no reason
CREATE TABLE user_profiles (
  user_id BIGINT PRIMARY KEY REFERENCES users(id),
  bio     TEXT  -- This could just be a column on users
);

-- BETTER: Just add the column
ALTER TABLE users ADD COLUMN bio TEXT;
```
